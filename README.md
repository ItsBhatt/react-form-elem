## Install

```bash
npm install --save react-form-elem
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-form-elem'
import 'react-form-elem/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
